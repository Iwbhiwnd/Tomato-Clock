# Tomato-Clock
A pomodoro timer based on Python, works in Windows.

# Install
pip install ft_tomato_clock

# How to use?
tomato           # Start 25-min focus + 5-min break
tomato -t        # Start 25-min focus
tomato -t <n>    # Start <n>-min focus
tomato -b        # Start 5-min break
tomato -b <n>    # Start <n>-min break
tomato -f        # Start unlimited focus (Ctrl+C to Exit)
tomato -all      # Show all-time total stats
tomato -clear    # Clear all log data
tomato -h        # Show help
